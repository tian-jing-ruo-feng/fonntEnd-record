let str: string = 'hello world'
let str1 = 'hello ts' // 推荐简写方式

// number 数字
let num: number = 55555
let num1 = 55555
let num: number

// boolean 布尔值
let boo: boolean = true
let boo1 = true
let boo2: boolean
let booo3 = `${boo}` // 支持模版字符串格式

// 数组

let arr3: Array<number> = [1, 2, 3] // 使用数组范型
